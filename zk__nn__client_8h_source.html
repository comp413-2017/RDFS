72 struct TargetDN {
73  char policy;
74  std::string dn_id;
75  uint64_t free_bytes; // free space on disk
76  uint32_t num_xmits; // current number of xmits
77 
78  TargetDN(std::string id, int bytes, int xmits, char policy) : policy(policy),
79  dn_id(id),
80  free_bytes(bytes),
81  num_xmits(xmits) {
82  }
83 
84  bool operator<(const struct TargetDN &other) const {
85  // If storage policy is 'x' for xmits, choose the min xmits node
86  if (policy == MIN_XMITS) {
87  if (num_xmits == other.num_xmits) {
88  return free_bytes < other.free_bytes;
89  }
90  return num_xmits > other.num_xmits;
91 
92  // Default policy is choose the node with the most free space
93  } else {
94  if (free_bytes == other.free_bytes) {
95  return num_xmits > other.num_xmits;
96  }
97  return free_bytes < other.free_bytes;
98  }
99  }
100 };

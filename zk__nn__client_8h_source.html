40 struct TargetDN {
41  char policy;
42  std::string dn_id;
43  uint64_t free_bytes; // free space on disk
44  uint32_t num_xmits; // current number of xmits
45 
46  TargetDN(std::string id, int bytes, int xmits, char policy) : dn_id(id),
47  free_bytes(bytes),
48  policy(policy),
49  num_xmits(xmits) {
50  }
51 
52  bool operator<(const struct TargetDN &other) const {
53  // If storage policy is 'x' for xmits, choose the min xmits node
54  if (policy == MIN_XMITS) {
55  if (num_xmits == other.num_xmits) {
56  return free_bytes < other.free_bytes;
57  }
58  return num_xmits > other.num_xmits;
59 
60  // Default policy is choose the node with the most free space
61  } else {
62  if (free_bytes == other.free_bytes) {
63  return num_xmits > other.num_xmits;
64  }
65  return free_bytes < other.free_bytes;
66  }
67  }
68 };

70 struct TargetDN {
71  char policy;
72  std::string dn_id;
73  uint64_t free_bytes; // free space on disk
74  uint32_t num_xmits; // current number of xmits
75 
76  TargetDN(std::string id, int bytes, int xmits, char policy) : policy(policy),
77  dn_id(id),
78  free_bytes(bytes),
79  num_xmits(xmits) {
80  }
81 
82  bool operator<(const struct TargetDN &other) const {
83  // If storage policy is 'x' for xmits, choose the min xmits node
84  if (policy == MIN_XMITS) {
85  if (num_xmits == other.num_xmits) {
86  return free_bytes < other.free_bytes;
87  }
88  return num_xmits > other.num_xmits;
89 
90  // Default policy is choose the node with the most free space
91  } else {
92  if (free_bytes == other.free_bytes) {
93  return num_xmits > other.num_xmits;
94  }
95  return free_bytes < other.free_bytes;
96  }
97  }
98 };

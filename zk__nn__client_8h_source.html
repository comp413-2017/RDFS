53 struct TargetDN {
54  char policy;
55  std::string dn_id;
56  uint64_t free_bytes; // free space on disk
57  uint32_t num_xmits; // current number of xmits
58 
59  TargetDN(std::string id, int bytes, int xmits, char policy) : dn_id(id),
60  free_bytes(bytes),
61  policy(policy),
62  num_xmits(xmits) {
63  }
64 
65  bool operator<(const struct TargetDN &other) const {
66  // If storage policy is 'x' for xmits, choose the min xmits node
67  if (policy == MIN_XMITS) {
68  if (num_xmits == other.num_xmits) {
69  return free_bytes < other.free_bytes;
70  }
71  return num_xmits > other.num_xmits;
72 
73  // Default policy is choose the node with the most free space
74  } else {
75  if (free_bytes == other.free_bytes) {
76  return num_xmits > other.num_xmits;
77  }
78  return free_bytes < other.free_bytes;
79  }
80  }

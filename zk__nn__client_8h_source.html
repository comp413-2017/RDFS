48 struct TargetDN {
49  char policy;
50  std::string dn_id;
51  uint64_t free_bytes; // free space on disk
52  uint32_t num_xmits; // current number of xmits
53 
54  TargetDN(std::string id, int bytes, int xmits, char policy) : dn_id(id),
55  free_bytes(bytes),
56  policy(policy),
57  num_xmits(xmits) {
58  }
59 
60  bool operator<(const struct TargetDN &other) const {
61  // If storage policy is 'x' for xmits, choose the min xmits node
62  if (policy == MIN_XMITS) {
63  if (num_xmits == other.num_xmits) {
64  return free_bytes < other.free_bytes;
65  }
66  return num_xmits > other.num_xmits;
67 
68  // Default policy is choose the node with the most free space
69  } else {
70  if (free_bytes == other.free_bytes) {
71  return num_xmits > other.num_xmits;
72  }
73  return free_bytes < other.free_bytes;
74  }
75  }
76 };
